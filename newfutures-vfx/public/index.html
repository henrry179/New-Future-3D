WebSocket API

实时通信和状态更新的WebSocket连接。

连接地址: ws://localhost:8000/ws
支持协议: vfx-v1
JavaScript 连接示例 
const ws = new WebSocket('ws://localhost:8000/ws', 'vfx-v1');

ws.onopen = function(event) {
    console.log('WebSocket连接已建立');
    // 订阅渲染进度更新
    ws.send(JSON.stringify({
        type: 'subscribe',
        channel: 'render_progress',
        task_id: 'render_task_67890'
    }));
};

ws.onmessage = function(event) {
    const data = JSON.parse(event.data);
    console.log('收到消息:', data);
    
    if (data.type === 'render_progress') {
        updateProgressBar(data.progress);
    }
};

API返回的标准错误代码和处理方式。

示例代码

常用场景的完整代码示例。

Python 示例

创建粒子特效并渲染 
import requests
import time

# API基础URL
BASE_URL = "http://localhost:8000/api/v1"
JWT_TOKEN = "your_jwt_token_here"

headers = {
    "Authorization": f"Bearer {JWT_TOKEN}",
    "Content-Type": "application/json"
}

# 1. 启动粒子特效
effect_config = {
    "parameters": {
        "count": 100000,
        "lifetime": 5.0,
        "size": 1.5,
        "color": [1.0, 0.6, 0.2],
        "gravity": -9.8
    }
}

response = requests.post(
    f"{BASE_URL}/vfx/effects/particles/start",
    json=effect_config,
    headers=headers
)
effect_session = response.json()["session_id"]

# 2. 开始渲染
render_config = {
    "resolution": "1920x1080",
    "quality": "high",
    "output_format": "mp4",
    "effects": ["particles"],
    "duration": 300,
    "fps": 30
}

response = requests.post(
    f"{BASE_URL}/render/start",
    json=render_config,
    headers=headers
)
task_id = response.json()["task_id"]

# 3. 监控渲染进度
while True:
    response = requests.get(
        f"{BASE_URL}/render/progress/{task_id}",
        headers=headers
    )
    progress_data = response.json()
    
    print(f"渲染进度: {progress_data['progress']}%")
    
    if progress_data["status"] == "completed":
        print("渲染完成!")
        break
    elif progress_data["status"] == "error":
        print("渲染失败!")
        break
    
    time.sleep(5)  # 每5秒检查一次

前端集成示例 
class VFXClient {
    constructor(baseUrl, token) {
        this.baseUrl = baseUrl;
        this.token = token;
        this.headers = {
            'Authorization': `Bearer ${token}`,
            'Content-Type': 'application/json'
        };
    }

    async startEffect(effectId, parameters) {
        const response = await fetch(
            `${this.baseUrl}/vfx/effects/${effectId}/start`,
            {
                method: 'POST',
                headers: this.headers,
                body: JSON.stringify({ parameters })
            }
        );
        return response.json();
    }

    async startRender(config) {
        const response = await fetch(
            `${this.baseUrl}/render/start`,
            {
                method: 'POST',
                headers: this.headers,
                body: JSON.stringify(config)
            }
        );
        return response.json();
    }

    async getProgress(taskId) {
        const response = await fetch(
            `${this.baseUrl}/render/progress/${taskId}`,
            { headers: this.headers }
        );
        return response.json();
    }
}

// 使用示例
const vfxClient = new VFXClient('http://localhost:8000/api/v1', 'your_token');

async function createParticleEffect() {
    try {
        // 启动粒子特效
        const effectResult = await vfxClient.startEffect('particles', {
            count: 50000,
            lifetime: 3.0,
            size: 2.0
        });
        
        console.log('特效已启动:', effectResult);
        
        // 开始渲染
        const renderResult = await vfxClient.startRender({
            resolution: '1920x1080',
            quality: 'medium',
            output_format: 'mp4',
            effects: ['particles']
        });
        
        console.log('渲染已开始:', renderResult);
        
        // 监控进度
        const taskId = renderResult.task_id;
        const progressInterval = setInterval(async () => {
            const progress = await vfxClient.getProgress(taskId);
            console.log(`进度: ${progress.progress}%`);
            
            if (progress.status === 'completed') {
                clearInterval(progressInterval);
                console.log('渲染完成!');
            }
        }, 2000);
        
    } catch (error) {
        console.error('操作失败:', error);
    }
}
